#include<stdio.h>
int FrogJumping(int n)
{
	if (n == 1)
		return 1;
	else if (n == 2)
		return 2;
	else
		return FrogJumping(n - 1) + FrogJumping(n - 2);
}
int main()
{
	/*
	一只青蛙要跳上n层高的台阶，一次能跳1/2阶，请问这只青蛙跳上n层高的台阶有多少种跳法？
	台阶数           跳法           怎样跳
	1               1         1
	2               2         1+1    2
	3               3         1+1+1  1+2   2+1
	4			    5         1+1+1+1  1+2+1  1+1+2  2+1+1 2+2
	5               8         1+1+1+1+1    1+2+2  2+2+1  2+1+2  2+1+1+1 1+2+1+1 1+1+2+1 1+1+1+2
	6               13        1+1+1+1+1+1  2+1+1+1+1（5） 2+2+1+1（6） 2+2+2
	7               21      1+1+1+1+1+1 2+1+1+1+1+1(6) 2+2+1+1+1(10) 2+2+2+1(4)
	这应该是递归问题
	好像是和斐波那契数列类型相同

	*/
	int n;
	printf("请输入青蛙要跳的台阶数：》");
	scanf_s("%d", &n);
	int x = FrogJumping(n);
	printf("青蛙要跳的次数为%d次\n", x);
		return 0;
}
